#include "mbed.h"

Ticker     t1;
Ticker     t2;
Ticker     t3;
Ticker     t4;
DigitalOut led1(LED1);
DigitalOut led2(LED2);
DigitalOut led3(LED3);
DigitalOut led4(LED4);

void
switch1s()
{
  led1 = !led1;
}

void
switch2s()
{
  led2 = !led2;
}

void
switch4s()
{
  led3 = !led3;
}

void
switch8s()
{
  led4 = !led4;
}

int
main()
{
  t1.attach(&switch1s, 1.0);
  t2.attach(&switch2s, 2.0);
  t3.attach(&switch4s, 4.0);
  t4.attach(&switch8s, 8.0);

  while (1) {
    printf("doin a whole lotta nothin...");
    wait(1);
  }
}
